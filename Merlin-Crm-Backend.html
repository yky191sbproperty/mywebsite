function getVisitorId() {
  var vid = localStorage.getItem('merlin_visitor_id');
  if (!vid) {
    vid = [navigator.language, screen.width+'x'+screen.height,
           new Date().getTimezoneOffset(), navigator.hardwareConcurrency||0
           ].join('|') + '_' + Date.now().toString(36);
    localStorage.setItem('merlin_visitor_id', vid);
  }
  return vid;
}
