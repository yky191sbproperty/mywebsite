try {
  var existing = JSON.parse(localStorage.getItem('merlin_crm_leads') || '[]');
  existing.push({
    id: 'sess_' + _sessionId + '_' + Date.now(),
    sessionId: _sessionId,
    time: new Date().toISOString(),
    msg: msg,
    flow: flowState ? flowState.key : 'chat',
    step: flowState ? flowState.step : 0,
    answers: flowState ? flowState.answers.slice() : [],
    lang: getLang(),
    status: 'new'
  });
  localStorage.setItem('merlin_crm_leads', JSON.stringify(existing));
} catch(e) {}
